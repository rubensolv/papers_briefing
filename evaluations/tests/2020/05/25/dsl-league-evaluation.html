#collapse-hide
dt9_8 = pd.read_csv("/home/rubens/papers_briefing/csvs/map9x8.csv")
#dt9_8.head()


#collapse-hide
selector = alt.selection_single(empty='all', fields=['Player'])
l = dt9_8['Player'].tolist()
color_scale = alt.Scale(domain=l,scheme="category20b")
base = alt.Chart(dt9_8).properties(
    width=800,
    height=200
).add_selection(selector)

points = base.mark_bar(filled=True, size=60).encode(
    x=alt.X('Player:N'),
    y=alt.Y('TOTAL:Q',
            scale=alt.Scale(domain=[0,max(dt9_8['TOTAL'].tolist())])),
    color=alt.condition(selector,
                        'Player:N',
                        alt.value('lightgray'),
                        scale=color_scale),
)

hists = base.mark_bar(opacity=0.5, thickness=100).encode(
    x=alt.X('Player:N'),
    y=alt.Y('STT:Q',
            stack=None,
            scale=alt.Scale(domain=[0,10])),
    color=alt.Color('Player:N',
                    scale=color_scale),
    tooltip=['Code:N']
).transform_filter(
    selector
).properties(
    width=160,
    height=200
)

hists2 = base.mark_bar(opacity=0.5, thickness=100).encode(
    x=alt.X('Player:N'),
    y=alt.Y('LIR:Q',
            stack=None,
            scale=alt.Scale(domain=[0,10])),
    color=alt.Color('Player:N',
                    scale=color_scale),
    tooltip=['Code:N']
).transform_filter(
    selector
).properties(
    width=160,
    height=200
)

hists3 = base.mark_bar(opacity=1.0, thickness=80).encode(
    x=alt.X('Player:N'),
    y=alt.Y('NS:Q',
            stack=None,
            scale=alt.Scale(domain=[0,10])),
    color=alt.Color('Player:N',
                    scale=color_scale),
    tooltip=['Code:N']
).transform_filter(
    selector
).properties(
    width=160,
    height=200
)

hists4 = base.mark_bar(opacity=1.0, thickness=80).encode(
    x=alt.X('Player:N'),
    y=alt.Y('A3N:Q',
            stack=None,
            scale=alt.Scale(domain=[0,10])),
    color=alt.Color('Player:N',
                    scale=color_scale),
    tooltip=['Code:N']
).transform_filter(
    selector
).properties(
    width=160,
    height=200
)

alt.vconcat(points , hists | hists2 | hists3 | hists4)
